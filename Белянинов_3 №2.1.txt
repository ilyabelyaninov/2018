using System;
using System.Collections.Generic;
using System.Linq;
using System.Text;
using System.Text.RegularExpressions;


namespace ConsoleApp1
{
    class Program
    {
        static void Main(string[] args)
        {
           
            string pattern = @"^(\+[7]\-\d{3}\-\d{3}\-\d{4})$";
            Regex myReg = new Regex(pattern);
            while (true)
            {
                Console.WriteLine("Введите свой номер телефона");
                string number = Console.ReadLine();

                if (Regex.IsMatch(number, pattern, RegexOptions.IgnoreCase))
                {
                    Console.WriteLine("Номер телефона подтвержден");
                    break;

                }
                else
                {
                    Console.WriteLine("Некорректный номер телефона");
                }
            }
            Console.ReadKey();
        }
    }
}
