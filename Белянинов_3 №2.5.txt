using System;
using System.Collections.Generic;
using System.Linq;
using System.Text;
using System.Text.RegularExpressions;


namespace ConsoleApp1
{
    class Program
    {
        static void Main(string[] args)
        {
           
            string pattern = @"^(\d{6}\,\sгород\s[А-Я]{1}[а-я]+\,\sулица\s[А-Я]{1}[а-я]+\,\sдом\s\d{1,3}\,\sквартира\s\d{1,3}\.)$";
            while (true)
            {
                Console.WriteLine("Введите свой адрес");
                string adress = Console.ReadLine();

                if (Regex.IsMatch(adress, pattern))
                {
                    Console.WriteLine("Адрес подтвержден");
                    break;
                }
                else
                {
                    Console.WriteLine("Некорректный адрес");
                }
            }
            Console.ReadKey();
        }
    }
}
